      function renderPixQr(payload){
      var el=document.getElementById("pix-qr");
      if(!el)return;
      el.innerHTML="";
      if(typeof QRCode!=="undefined"){
      new QRCode(el,{text:payload,width:180,height:180,colorDark:"#000000",colorLight:"#ffffff",correctLevel:QRCode.CorrectLevel.M});
      }else{
      var sc=document.createElement("script");
      sc.src="https://cdnjs.cloudflare.com/ajax/libs/qrcodejs/1.0.0/qrcode.min.js";
      sc.onload=function(){new QRCode(el,{text:payload,width:180,height:180,colorDark:"#000000",colorLight:"#ffffff",correctLevel:QRCode.CorrectLevel.M});};
      document.head.appendChild(sc);
      }
      }
